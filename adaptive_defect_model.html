def adapdef(y,t):
    
    z=np.zeros(np.size(y))

    z[0]=pv*y[2]-dv*y[0]-B*y[0]*y[1]
    
    z[1]=gr*(y[1]+y[3])*(1-(y[1]+y[3]+y[2])/T0)-B2*y[0]*y[1]+rho*y[3]-phi*y[4]*y[1]
    z[2]=B2*y[0]*y[1]-di*y[2]-kn*y[2]*y[4]
    z[3]=phi*y[4]*y[1]-rho*y[3] ##R
    z[4]=pf*y[2]-df*y[4] ##F

    return z
